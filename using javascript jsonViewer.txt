        function view() {
            try {
                var data = eval('(' + $('#json-input').val() + ')'); // convert json string to object
            }
            catch (ex) {
                data = '';
            }
            
            var options = { collapsed: false, withQuotes: true };
            $('#json-renderer').jsonViewer(data, options);
        }

        $(function () {
            $('#btn-json-viewer').click(view);
