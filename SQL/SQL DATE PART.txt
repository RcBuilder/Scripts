SQL DATE PART
-------------

SELECT DATEPART(WEEKDAY, GETDATE())  -- 5
SELECT DATENAME(WEEKDAY, GETDATE())  -- Thursday

SELECT DAY(GETDATE())  -- 28
SELECT MONTH(GETDATE())  -- 11
SELECT YEAR(GETDATE())  -- 2019

---

// get current week
DECLARE @TODAY DATE = GETDATE();	
DECLARE @SUNDAY DATE = (SELECT DATEADD(DAY,  -(DATEPART(WEEKDAY, @TODAY)) + 1, @TODAY))  
SELECT  @SUNDAY, @TODAY
